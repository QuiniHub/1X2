#!/usr/bin/env python3
from pathlib import Path
import csv, json

ROOT = Path(__file__).resolve().parents[1]
DATA = ROOT / "data"

rows = list(csv.DictReader((DATA / "jornada_61_full.csv").open(encoding="utf-8")))
out = {
    "jornada": 61,
    "pronostico": [
        {
            "n": r["n"],
            "partido": f'{r["local"]} - {r["visitante"]}',
            "signo": r["signo"],
            "tipo": r["tipo"],
            "indice": r["indice_sorpresa"]
        }
        for r in rows
    ]
}
(DATA / "pronostico_actual.json").write_text(json.dumps(out, indent=2, ensure_ascii=False), encoding="utf-8")
print("Pronóstico actual generado.")
